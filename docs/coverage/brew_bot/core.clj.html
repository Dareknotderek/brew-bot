001  (ns brew-bot.core
002    "Main fns to build beer recipes"
003    (:require [brew-bot.ingredients :as ingredients]
004              [clojure.set :as cset])
005    (:gen-class))
006  
007  (defn update-or-assoc
008    "If `key` exists in `map` apply the `update-fn`.
009     Else, assoc `val` to that `key`"
010    [map key val update-fn]
011    (if (get map key)
012      (update map key update-fn)
013      (assoc map key val)))
014  
015  (defn scale-ingredients
016    [ingredient-map weight-limit current-weight]
017    (loop [i-map ingredient-map
018           weight current-weight]
019      (if (< weight weight-limit)
020        (let [addition (rand-nth ingredients/ingredient-amounts)
021              mod-ingredient (rand-nth (keys i-map))]
022          (recur (update i-map mod-ingredient + addition)
023                 (+ weight addition)))
024        i-map)))
025  
026  (defn generate-ingredients-and-quantities
027    [ingredient-set weight-cutoff ingredient-limit]
028    (loop [ingredient-map {}
029           weight 0.0
030           ingredient-count 0]
031      (let [selected-ingredient (rand-nth (keys ingredient-set))
032            ingredient-addition (rand-nth ingredients/ingredient-amounts)]
033        (if (< ingredient-count ingredient-limit)
034          (if (< weight weight-cutoff)
035            (recur (update-or-assoc ingredient-map selected-ingredient ingredient-addition #(+ ingredient-addition %))
036                   (+ weight ingredient-addition)
037                   (count (keys ingredient-map)))
038            ingredient-map)
039          (scale-ingredients ingredient-map weight-cutoff weight)))))
040  
041  (defn generate-beer-recipe
042    [grain-weight-limit grain-item-limit extract-weight-limit extract-item-limit hop-weight-limit hop-item-limit]
043    (let [grain-bill   (generate-ingredients-and-quantities ingredients/base-grains grain-weight-limit   grain-item-limit)
044          extract-bill (generate-ingredients-and-quantities ingredients/extracts    extract-weight-limit extract-item-limit)
045          hop-bill     (generate-ingredients-and-quantities ingredients/hops        hop-weight-limit     hop-item-limit)
046          yeast        (rand-nth (keys ingredients/yeasts))
047          _ (println grain-bill)
048          _ (println extract-bill)
049          _ (println (map #(str (first %) " " (second %) "oz " (rand-nth ingredients/hop-times)) hop-bill))
050          _ (println yeast)]
051      :placeholder))
052  
053  (defn -main
054    "Try me out!"
055    [& args]
056    (generate-beer-recipe 10.0 3 5.0 1 1.5 2))
