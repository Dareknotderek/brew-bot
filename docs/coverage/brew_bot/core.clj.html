001  (ns brew-bot.core
002    "Main fns to build beer recipes"
003    (:require [brew-bot.generators :as generators])
004    (:gen-class))
005  
006  (defn generate-beer-recipe
007    [strategy gallons grain-limits extract-limits hop-limits yeast-limits]
008    (case strategy
009      :random (generators/generate-random-recipe gallons (:weight grain-limits) (:weight extract-limits) (:weight hop-limits))
010      :limited-random (generators/generate-limited-random-recipe gallons (:weight grain-limits) (:count grain-limits) (:weight extract-limits) (:count extract-limits) (:weight hop-limits) (:count hop-limits))
011      :weighted-guided (generators/generate-weighted-guided-recipe gallons grain-limits extract-limits hop-limits yeast-limits)
012      {:invalid-strategy "Please pick a legal recipe generation strategy"}))
013  
014  (defn -main
015    "Try me out!"
016    [& args]
017    (println (generate-beer-recipe :weighted-guided 5 {:weight 5.0 :probabilities {:pale-malt-2-row-us 40 :pale-malt-2-row-uk 40 :caramel-crystal-malt-40l 20 :caramel-crystal-malt-60l 10 :chocolate-malt 2 :black-patent-malt 1}}
018                                   {:weight 3.5 :probabilities {:amber-liquid-extract 1 :light-dry-extract 1}}
019                                   {:weight 3.0 :probabilities {:magnum-us 2 :willamette 4 :liberty 1 :cascade 1}}
020                                   {:probabilities {:wyeast-labs-european-ale 1 :wyeast-labs-irish-ale 1 :wyeast-labs-american-ale 1 :wyeast-labs-belgian-saison 1}})))
