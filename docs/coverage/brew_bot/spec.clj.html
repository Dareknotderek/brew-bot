001  (ns brew-bot.spec
002    "Common specs/api checks for core functions"
003    (:require [brew-bot.ingredients :as ingredients]
004              [clojure.spec.alpha :as s]))
005  
006  (s/def ::name string?)
007  (s/def ::product-number string?)
008  (s/def ::manufacturer string?)
009  
010  (s/def ::gravity
011    (s/and number?
012           #(>= % 1.0)))
013  
014  (s/def ::alpha
015    (s/and number?
016           #(>= % 1.0)))
017  
018  (s/def ::beta
019    (s/and number?
020           #(>= % 1.0)))
021  
022  (s/def ::weight
023    (s/and number?
024           #(>= % 0.25)))
025  
026  (s/def ::grain
027    (s/keys :req-un [::name ::gravity]
028            :opt-un [::weight]))
029  
030  (s/def ::grains
031    (s/map-of ingredients/grains-keys ::grain))
032  
033  (s/def ::hop
034    (s/keys :req-un [::name ::alpha ::beta]
035            :opt-un [::weight]))
036  
037  (s/def ::hops
038    (s/map-of ingredients/hops-keys ::hop))
039  
040  (s/def ::extract
041    (s/keys :req-un [::name ::gravity]
042            :opt-un [::weight]))
043  
044  (s/def ::extracts
045    (s/map-of ingredients/extracts-keys ::extract))
046  
047  (s/def ::yeast
048    (s/keys :req-un [::name ::product-number ::manufacturer]))
049  
050  (s/def ::yeasts
051    (s/map-of ingredients/yeasts-keys ::yeast))
052  
053  (s/def ::recipe
054    (s/keys :req-un [::grains ::hops ::yeasts ::gravity]
055            :opt-un [::extracts]))
