001  (ns brew-bot.generators
002    "Beer recipe generators"
003    (:require [brew-bot.ingredients :as ingredients]
004              [brew-bot.util :as util]))
005  
006  (defn format-recipe
007    "Given maps of ingredient : weight pairs, format a recipe and derive other important information"
008    [gallons grain-bill extract-bill hop-bill yeast]
009    (let [grains   (util/join-ingredient-maps grain-bill   ingredients/base-grains)
010          extracts (util/join-ingredient-maps extract-bill ingredients/extracts)
011          hops     (util/join-ingredient-maps hop-bill     ingredients/hops)]
012      {:grains   grains
013       :extracts extracts
014       :hops     hops
015       :yeast    yeast
016       :original-gravity (util/calculate-gravity gallons grains extracts)}))
017  
018  (defn generate-ingredients-and-quantities
019    ([ingredient-set weight-cutoff]
020     (generate-ingredients-and-quantities ingredient-set weight-cutoff :unlimited))
021  
022    ([ingredient-set weight-cutoff ingredient-limit]
023     (loop [ingredient-map {}
024            weight 0.0
025            ingredient-count 0]
026       (if (or (= ingredient-limit :unlimited)
027               (< ingredient-count ingredient-limit))
028         (if (< weight weight-cutoff)
029           (let [selected-ingredient (util/rand-key ingredient-set)
030                 ingredient-addition (rand-nth ingredients/ingredient-amounts)
031                 updated-map (util/update-or-assoc ingredient-map selected-ingredient ingredient-addition #(+ ingredient-addition %))]
032             (recur updated-map
033                    (+ weight ingredient-addition)
034                    (count (keys updated-map))))
035           ingredient-map)
036         (util/scale-ingredients ingredient-map weight-cutoff)))))
037  
038  (defn generate-random-recipe
039    "Generate a purely random beer recipe with grains, extracts, and hops close to their respective limits"
040    [gallons grain-weight-limit extract-weight-limit hop-weight-limit]
041    (let [grain-bill   (generate-ingredients-and-quantities ingredients/base-grains grain-weight-limit)
042          extract-bill (generate-ingredients-and-quantities ingredients/extracts    extract-weight-limit)
043          hop-bill     (generate-ingredients-and-quantities ingredients/hops        hop-weight-limit)
044          yeast        (util/rand-key ingredients/yeasts)]
045      (format-recipe gallons grain-bill extract-bill hop-bill yeast)))
046  
047  (defn generate-limited-random-recipe
048    "Generate a purely random beer recipe with grains, extracts,
049     and hops close to their respective weight limits, with a constricted number of ingredients"
050    [gallons grain-weight-limit grain-item-limit extract-weight-limit extract-item-limit hop-weight-limit hop-item-limit]
051    (let [grain-bill   (generate-ingredients-and-quantities ingredients/base-grains grain-weight-limit   grain-item-limit)
052          extract-bill (generate-ingredients-and-quantities ingredients/extracts    extract-weight-limit extract-item-limit)
053          hop-bill     (generate-ingredients-and-quantities ingredients/hops        hop-weight-limit     hop-item-limit)
054          yeast        (util/rand-key ingredients/yeasts)]
055      (format-recipe gallons grain-bill extract-bill hop-bill yeast)))
