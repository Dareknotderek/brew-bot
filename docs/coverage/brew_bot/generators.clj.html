001  (ns brew-bot.generators
002    "Beer recipe generators"
003    (:require [bigml.sampling.simple :as bss]
004              [brew-bot.ingredients :as ingredients]
005              [brew-bot.util :as util]))
006  
007  (defn format-recipe
008    "Given maps of ingredient : weight pairs, format a recipe and derive other important information"
009    [gallons grain-bill extract-bill hop-bill yeast]
010    (let [grains   (util/join-ingredient-maps grain-bill ingredients/base-grains :weight)
011          extracts (util/join-ingredient-maps extract-bill ingredients/extracts :weight)
012          hops     (util/join-ingredient-maps hop-bill ingredients/hops :weight)]
013      {:grains   grains
014       :extracts extracts
015       :hops     hops
016       :yeast    yeast
017       :original-gravity (util/calculate-gravity gallons grains extracts)}))
018  
019  (defn update-selection-probability
020    [probabilities ingredient-map include-all?]
021    (let [weighted-ingredients (util/join-ingredient-maps probabilities ingredient-map :probability)]
022      (if include-all?
023        (merge ingredient-map weighted-ingredients)
024        weighted-ingredients)))
025  
026  (defn generate-ingredients-and-quantities
027    ([ingredient-set weight-cutoff]
028     (generate-ingredients-and-quantities ingredient-set weight-cutoff :unlimited))
029  
030    ([ingredient-set weight-cutoff ingredient-limit]
031     (loop [ingredient-map {}
032            weight 0.0
033            ingredient-count 0]
034       (if (or (= ingredient-limit :unlimited)
035               (< ingredient-count ingredient-limit))
036         (if (< weight weight-cutoff)
037           (let [selected-ingredient (util/rand-key ingredient-set)
038                 ingredient-addition (first (bss/sample ingredients/ingredient-amounts :replace true))
039                 updated-map (util/update-or-assoc ingredient-map selected-ingredient ingredient-addition #(+ ingredient-addition %))]
040             (recur updated-map
041                    (+ weight ingredient-addition)
042                    (count (keys updated-map))))
043           ingredient-map)
044         (util/scale-ingredients ingredient-map weight-cutoff)))))
045  
046  (defn generate-random-recipe
047    "Generate a purely random beer recipe with grains, extracts, and hops close to their respective limits"
048    [gallons grain-weight-limit extract-weight-limit hop-weight-limit]
049    (let [grain-bill   (generate-ingredients-and-quantities ingredients/base-grains grain-weight-limit)
050          extract-bill (generate-ingredients-and-quantities ingredients/extracts    extract-weight-limit)
051          hop-bill     (generate-ingredients-and-quantities ingredients/hops        hop-weight-limit)
052          yeast        (util/rand-key ingredients/yeasts)]
053      (format-recipe gallons grain-bill extract-bill hop-bill yeast)))
054  
055  (defn generate-limited-random-recipe
056    "Generate a purely random beer recipe with grains, extracts,
057     and hops close to their respective weight limits, with a constricted number of ingredients"
058    [gallons grain-weight-limit grain-item-limit extract-weight-limit extract-item-limit hop-weight-limit hop-item-limit]
059    (let [grain-bill   (generate-ingredients-and-quantities ingredients/base-grains grain-weight-limit   grain-item-limit)
060          extract-bill (generate-ingredients-and-quantities ingredients/extracts    extract-weight-limit extract-item-limit)
061          hop-bill     (generate-ingredients-and-quantities ingredients/hops        hop-weight-limit     hop-item-limit)
062          yeast        (util/rand-key ingredients/yeasts)]
063      (format-recipe gallons grain-bill extract-bill hop-bill yeast)))
064  
065  (defn generate-weighted-guided-recipe
066    [gallons grain-limits extract-limits hop-limits yeast-limits]
067    (let [grain-selections   (update-selection-probability (:probabilities grain-limits) ingredients/base-grains false)
068          extract-selections (update-selection-probability (:probabilities extract-limits) ingredients/extracts false)
069          hop-selections     (update-selection-probability (:probabilities hop-limits) ingredients/hops false)
070          yeast-selections   (update-selection-probability (:probabilities yeast-limits) ingredients/yeasts false)
071          grain-bill   (generate-ingredients-and-quantities grain-selections (:weight grain-limits))
072          extract-bill (generate-ingredients-and-quantities extract-selections (:weight extract-limits))
073          hop-bill     (generate-ingredients-and-quantities hop-selections (:weight hop-limits))
074          yeast        (first (bss/sample yeast-selections))]
075      (format-recipe gallons grain-bill extract-bill hop-bill yeast)))
